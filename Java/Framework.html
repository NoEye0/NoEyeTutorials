3. Angular

Angular es un framework robusto para construir aplicaciones web. Debido a la complejidad de Angular, el ejemplo básico se realiza usando un entorno local de desarrollo.

Ejemplo básico de Angular:

                <!-- index.html -->
                <!DOCTYPE html>
                <html>
                <head>
                    <meta charset="UTF-8">
                    <title>Angular Example</title>
                    <script src="https://unpkg.com/@angular/core/bundles/core.umd.js"></script>
                    <script src="https://unpkg.com/@angular/common/bundles/common.umd.js"></script>
                    <script src="https://unpkg.com/@angular/compiler/bundles/compiler.umd.js"></script>
                    <script src="https://unpkg.com/@angular/platform-browser/bundles/platform-browser.umd.js"></script>
                    <script src="https://unpkg.com/@angular/platform-browser-dynamic/bundles/platform-browser-dynamic.umd.js"></script>
                </head>
                <body>
                    <my-app></my-app>
                    <script>
                        (function (ng) {
                            var AppComponent = ng.core.Component({
                                selector: 'my-app',
                                template: '<h1>Hola, Angular!</h1>'
                            }).Class({
                                constructor: function () { }
                            });

                            ng.platformBrowserDynamic.platformBrowserDynamic().bootstrapModule(
                                ng.core.NgModule({
                                    imports: [ng.platformBrowser.BrowserModule],
                                    declarations: [AppComponent],
                                    bootstrap: [AppComponent]
                                }).Class({
                                    constructor: function () { }
                                })
                            );
                        })(window.ng);
                    </script>
                </body>
                </html>

Nota: Angular requiere un entorno más elaborado para trabajar, por lo que el ejemplo está simplificado para este contexto.
